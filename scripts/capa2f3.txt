    if pa is None or pb is None or ga is None or gb is None:
        return False, False
    try:
        d_a_g = pa.DistanceTo(ga)
        d_b_g = pb.DistanceTo(ga)
        d_a_h = pa.DistanceTo(gb)
        d_b_h = pb.DistanceTo(gb)
    except Exception:
        return False, False
    ini_es_a = d_a_g <= d_b_g
    if ini_es_a:
        ti = bool(troceo_ini_glob) and d_a_g <= tol_ft
        tf = bool(troceo_fin_glob) and d_b_h <= tol_ft
    else:
        ti = bool(troceo_ini_glob) and d_b_g <= tol_ft
        tf = bool(troceo_fin_glob) and d_a_h <= tol_ft
    return ti, tf


def _subdividir_curva_por_cortes_interiores(curva, cortes_params):
    if curva is None:
        return []
    lo, hi = _rango_nativo_curva(curva)
    if lo is None or hi is None:
        return []
    cortes = _dedupe_sorted_params(list(cortes_params or []), _TOL_DEDUPE_PARAM_FT)
    cortes = _params_interiores(cortes, lo, hi)
    bounds = [lo] + list(cortes) + [hi]
    segmentos = []
    for i in range(len(bounds) - 1):
        a, b = bounds[i], bounds[i + 1]
        if b - a <= _MIN_LINE_LEN_FT * 2:
            continue
        seg = _subcurva_entre_parametros_nativos(curva, a, b)
        if seg is not None:
            segmentos.append(seg)
    return segmentos
